set SERVEROUTPUT ON
DECLARE
  MSISDN VARCHAR2(200);
  CIRCLE_ID VARCHAR2(200);
  RED_FAMILY_DTL KRISHNAKANT.RED_FAMILY_DETAILS_ASSET;
  ERROR_CODE VARCHAR2(200);
  ERROR_DESC VARCHAR2(200);
  
BEGIN
  DBMS_OUTPUT.ENABLE;
  MSISDN := '123';
  CIRCLE_ID := '123';

  GET_REDFAMILY_DETAILS(
    MSISDN => MSISDN,
    CIRCLE_ID => CIRCLE_ID,
    RED_FAMILY_DTL => RED_FAMILY_DTL,
    ERROR_CODE => ERROR_CODE,
    ERROR_DESC => ERROR_DESC
  );
   
DBMS_OUTPUT.PUT_LINE('RED_FAMILY_DTL = ' || RED_FAMILY_DTL.SIEB_CIRCLE_ID);
DBMS_OUTPUT.PUT_LINE('ERROR_CODE = ' || ERROR_CODE);
DBMS_OUTPUT.PUT_LINE('ERROR_DESC = ' || ERROR_DESC);

END;
